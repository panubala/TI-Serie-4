\begin{homeworkProblem}
	\begin{enumerate}[(a)]
		\item Wir verwenden die Methode der Kolmogorov-Komplexitaet um zu zeigen das $L_1$ nicht regulaer ist. Sei $L_1 =\{ 0^{\binom{2n}{n} }| n \in \mathbb{N}\}$. \textit{Angenommen} $L_1$ sei regulaer. Fuer jedes $m\in \mathbb{N} $ gilt, dass $\binom{2(m+1)}{(m+1)}-\binom{2m}{m} = \frac{(2m)!(m+1)(-m+1)}{((m+1)!)^2}$. $0^\frac{(2m)!(m+1)(-m+1)}{((m+1)!)^2} -1$ ist das erste Wort in der Sprache 
		$$L_{\binom{2m}{m}+1} =\{y| 0^{\binom{2m}{m}+1} y \in L\}$$
		
		Satz 3.1 folgend existiert eine Konstante c, die unabhaengig von x = $0^{\binom{2m}{m}+1}$ und y = $0^ {\frac{(2m)!(m+1)(-m+1)}{((m+1)!)^2-1}} $ und somit von m ist, so dass
		
		$$K(0^ {\frac{(2m)!(m+1)(-m+1)}{((m+1)!)^2}-1}) \leq \lceil log_2 (1+1)\rceil + c = 1+c$$
		
		Dies ist ein Widerspruch. Es gibt nur endlich viele Programme der konstanten Laenge hoechstens 1+c gibt, aber es gibt unendlich viele Woerter der Form $0^ {\frac{(2m)!(m+1)(-m+1)}{((m+1)!)^2-1}}$. Also ist die Annahme falsch und $L_1$ ist nicht regulaer.
		
		
		
	 
	\end{enumerate}
	
	\begin{enumerate}[(b)]
		\item
		\begin{enumerate}[]
		\item Wir zeigen indirekt, dass, $L_2  \not\in L_{EA}$.\\
			\textit{Annahme:} $L$ sei regulaer. Sei $A = (Q, \Sigma,\delta_A, q_0, F)$ ein EA mit L(A) = L.\\
		
		\item Wir betrachten die Woerter\\
			$$ b^1, b^2, ... 			b^{|Q| +1}$$
			
			Weil die Anzahl dieser Woerter $|Q| + 1$ ist, existieren $i,j \in \{1,2,..., |Q| + 1\}, i<j$, so dass \\
			$$\hat \delta (q_0, b_i) = \hat \delta (q_0, b_j)$$
			
		\item Nach Lemma 3.3 gilt
			$b^iz \in L  \Longleftrightarrow b^jz \in L $\\
			fuer alle $z \in \Sigma^*$. Dies gilt aber nicht , weil $z=a^{2i}$ das Wort $b^ia{^2i} \in L$ und das Wort $b^ja^{2i}\not\in L				$.\\
		
		
		\item Weil $j>i$
		\item $ \Rightarrow 2j > 2i$
		\item $\Rightarrow 2j > |w|_a$
		\item $\Rightarrow |w|_b > |w|_a$
		\item $\Rightarrow w \not\in L$
		
		\item Also ist die Annahme falsch und $L_2$ ist nicht regulaer
		\end{enumerate}
	\end{enumerate}
	
	
\end{homeworkProblem}

\begin{homeworkProblem}
	\begin{enumerate}[(a)]
	\item
		\begin{enumerate}[]
		\item Wir fuehren ein Widerspruchsbeweis. 
		\item \textit{Annahme}: L ist regulaer. Dann ist das \textit{Pumping Lemma} anwendbar. 
		Es existiert eine Konstante $n_0$ mit den in \textit{Lemma 3.4} beschriebenen Eigenschaften. Wir betrachten das Wort 
		$$w= 0^{n_0}1^{n_0}.$$
		Offensichtlich gilt $|w| = 2n_0 \geq n_0$. Also muss eine Zerlegung w = yxz von w geben, die die Bedingungen (i), (ii) und (iii) 			erfuellt. Wegen (i) gilt $|yx| \leq n_0$ also ist 
			\begin{center}
				$y = 0^a$\\
				$x = 0^b$\\
				$z = 0^{n_0 -a -b}1^{n_0}$ \\
			\end{center}
		fuer irgendwelche $a,b\in \mathbb{N}$. Wegen (ii) gilt $b > 0$. Da $w \notin L_3$ gilt nach (iii)
		
			$$\{yx^kz| k \in \mathbb{N}\} =\{ 0^{n_0 + (k-1)b}1^{n_0}| k \in \mathbb{N}\} \cap \emptyset $$
			
		Dies ist aber ein Widerspruch. Fuer k = 2 gilt $yx^2z =0^{n_0 +kb}1^{n_0} \in L_3$. Also ist die Annahme falsch und $L_3$ ist 			nicht regulaer
		
		
		\end{enumerate}
	\end{enumerate}
	
	\begin{enumerate}[(b)]
	\item
		\begin{enumerate}[]
			\item Wir wollen zegen, dass die nichtregulaere Sprache L die Bedingungen des abgeschwaechte Pumping-Lemmas erfuellt.\\ Es sei $L:= \{w \in \{0,1\}^* | |w|_0 = |w|_1 \}$. Es existiert eine Konstante $n_0$. Alle Woerter in der Sprache mit einer Laenge von mindestens $n_0$ muss eine Zerlegung besitzen, die die Eigenschaften (i'), (ii), (iii') erfuellt.\\
			Woerter, die 01 oder 10 nicht als Infix haben sind nicht in der Sprache enthalten. Nach der abgeschwaechte Version des Pumping-Lemmas muessen wir nur die Woerter $w\in L$ betrachten.
			\item Besitzt w hingegen 01 oder 10 als Infix, hat w die Form yxz. Dabei ist $x\in\{01,10\}$. Da wir $n_0 := 2$ waelen, sind die Bedingungen (i') und (ii) erfuellt. Denn es gilt $1\leq x \leq n_0$.\\
			Die dritte Bedingung $\{yx^kz| k\in \mathbb{N}\} \subseteq ist auch erfuellt denn es gilt:$
			
			\begin{center}
				$|yx^kz|_a-|yx^kz|_b$\\
				=$(|y|_a + |x^k|_a+ |z|_a) - (|y|_b + |x^k|_b+ |z|_b)$\\
				=$(|y|_a + k*|x|_a+ |z|_a) - (|y|_b + k*|x|_b+ |z|_b)$\\
				=$(|y|_a + k+ |z|_a) - (|y|_b + k+ |z|_b)$\\
				=$(|yz|_a + k) - (|yz|_b + k)$\\
				=$(|w|_a + (k-1)) - (|w|_b + (k-1))$\\
				=$(|w|_a) - (|w|_b)$\\
			\end{center}
			Da w in der Sprache ist, ist diese Zahl gleich null.

		
		\end{enumerate}
	\end{enumerate}
\end{homeworkProblem}
